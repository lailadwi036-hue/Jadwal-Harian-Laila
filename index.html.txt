    <script>
        // Set Current Date
        const dateOptions = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
        document.getElementById('current-date').innerText = new Date().toLocaleDateString('id-ID', dateOptions);

        // State Management
        let tasks = JSON.parse(localStorage.getItem('laila_tasks')) || [
            { id: 1, time: "07:00", text: "Bangun, Mandi, & Skincare-an 🧴", category: "Rutinitas", done: false },
            { id: 2, time: "16:00", text: "Nonton Drakor / Main Game 🎮", category: "Hobi", done: false }
        ];

        // Categories Colors mapping
        const categoryColors = {
            "Rutinitas": "bg-blue-100 text-blue-600 border-blue-200",
            "Hobi": "bg-pink-100 text-pink-600 border-pink-200",
            "Belajar": "bg-purple-100 text-purple-600 border-purple-200",
            "Healing": "bg-mint-100 text-emerald-600 border-emerald-200",
            "Lainnya": "bg-gray-100 text-gray-600 border-gray-200"
        };

        // DOM Elements
        const taskListEl = document.getElementById('task-list');
        const timeInput = document.getElementById('task-time');
        const catInput = document.getElementById('task-category');
        const textInput = document.getElementById('task-text');
        const statsEl = document.getElementById('task-stats');

        // Allow pressing Enter to add task
        textInput.addEventListener('keypress', function (e) {
            if (e.key === 'Enter') {
                addTask();
            }
        });

        function saveAndRender() {
            // Sort tasks by time
            tasks.sort((a, b) => a.time.localeCompare(b.time));
            // Save to Local Storage
            localStorage.setItem('laila_tasks', JSON.stringify(tasks));
            // Render to DOM
            renderTasks();
            updateStats();
        }

        function renderTasks() {
            taskListEl.innerHTML = '';

            if (tasks.length === 0) {
                taskListEl.innerHTML = `
                    <div class="text-center py-6 text-gray-400 font-medium">
                        <i class="fa-solid fa-face-smile-wink text-4xl mb-3 text-brand-purple opacity-50"></i>
                        <p>Belum ada jadwal nih. Yuk tambah kegiatan Laila!</p>
                    </div>
                `;
                return;
            }

            tasks.forEach(task => {
                const colors = categoryColors[task.category] || categoryColors["Lainnya"];
                
                const taskEl = document.createElement('div');
                taskEl.className = `task-item flex items-center gap-3 p-3 sm:p-4 rounded-2xl bg-white/70 border border-white shadow-sm ${task.done ? 'opacity-60' : ''}`;
                
                taskEl.innerHTML = `
                    <input type="checkbox" class="cute-checkbox flex-shrink-0" ${task.done ? 'checked' : ''} onchange="toggleTask(${task.id})">
                    
                    <div class="flex-1 min-w-0">
                        <div class="flex items-center gap-2 mb-1">
                            <span class="text-xs font-bold px-2 py-0.5 rounded-md border ${colors}">
                                ${task.category}
                            </span>
                            <span class="text-xs font-bold text-gray-500 bg-gray-100/80 px-2 py-0.5 rounded-md">
                                <i class="fa-regular fa-clock mr-1"></i>${task.time || '--:--'}
                            </span>
                        </div>
                        <p class="font-semibold text-gray-700 truncate ${task.done ? 'line-through text-gray-400' : ''}">
                            ${task.text}
                        </p>
                    </div>

                    <button onclick="deleteTask(${task.id})" class="flex-shrink-0 w-8 h-8 flex justify-center items-center rounded-full bg-red-50 text-red-400 hover:bg-red-100 hover:text-red-500 transition-colors">
                        <i class="fa-solid fa-trash-can text-sm"></i>
                    </button>
                `;
                taskListEl.appendChild(taskEl);
            });
        }

        function addTask() {
            const time = timeInput.value;
            const category = catInput.value;
            const text = textInput.value.trim();

            if (!text) {
                // Simple animation to shake input if empty
                textInput.classList.add('animate-pulse', 'border-red-400', 'border');
                setTimeout(() => textInput.classList.remove('animate-pulse', 'border-red-400', 'border'), 1000);
                return;
            }

            const newTask = {
                id: Date.now(), // unique ID based on timestamp
                time: time,
                category: category,
                text: text,
                done: false
            };

            tasks.push(newTask);
            
            // Reset input fields
            textInput.value = '';
            timeInput.value = '';
            
            saveAndRender();
        }

        function toggleTask(id) {
            const task = tasks.find(t => t.id === id);
            if (task) {
                task.done = !task.done;
                saveAndRender();
            }
        }

        function deleteTask(id) {
            tasks = tasks.filter(t => t.id !== id);
            saveAndRender();
        }

        function updateStats() {
            const total = tasks.length;
            const done = tasks.filter(t => t.done).length;
            const percentage = total === 0 ? 0 : Math.round((done / total) * 100);
            
            statsEl.innerText = `${done} dari ${total} selesai (${percentage}%)`;
        }

        // Initial Render
        saveAndRender();

    </script>
